30class PacificPokerSummary(TourneySummary): 

31 """Pacific Poker tournament summary parser.""" 

32 

33 limits: ClassVar = {"No Limit": "nl", "Pot Limit": "pl", "Limit": "fl", "LIMIT": "fl"} 

34 

35 games: ClassVar = { # base, category 

36 "Hold'em": ("hold", "holdem"), 

37 "Holdem": ("hold", "holdem"), 

38 "Omaha": ("hold", "omahahi"), 

39 "Omaha Hi/Lo": ("hold", "omahahilo"), 

40 "OmahaHL": ("hold", "omahahilo"), 

41 } 

42 

43 substitutions: ClassVar = { 

44 "LEGAL_ISO": "USD|EUR|GBP|CAD|FPP", # legal ISO currency codes 

45 "LS": r"\$|€|", # legal currency symbols 

46 "NUM": ".,\\d\xa0", # legal characters in number format 

47 } 

48 

49 re_identify = re.compile( 

50 r"\*{5}\s(Cassava|888poker|888)(\.[a-z]{2})?(\-[a-z]{2})? Tournament Summary\s\*{5}", 

51 ) 

52 

53 re_tourney_info = re.compile( 

54 r""" 

55 Tournament\sID:\s(?P<TOURNO>[0-9]+)\s+ 

56 Buy-In:\s(?P<BUYIN>(((?P<BIAMT>(?P<CURRENCY1>{LS})?[{NUM}]+\s?(?P<CURRENCY2>{LS})?)(\s\+\s?(?P<BIRAKE>({LS})?[{NUM}]+\s?({LS})?))?)|(Free)|(.+?)))\s+ 

57 (Rebuy:\s[{LS}](?P<REBUYAMT>[{NUM}]+)\s?({LS})?\s+)? 

58 (Add-On:\s[{LS}](?P<ADDON>[{NUM}]+)\s?({LS})?\s+)? 

59 ((?P<P1NAME>.*?)\sperformed\s(?P<PREBUYS>\d+)\srebuys?\s+)? 

60 ((?P<P2NAME>.*?)\sperformed\s(?P<PADDONS>\d+)\sadd-ons?\s+)? 

61 ^(?P<PNAME>.+?)\sfinished\s(?P<RANK>[0-9]+)\/(?P<ENTRIES>[0-9]+)(\sand\swon\s(?P<WCURRENCY>[{LS}])?(?P<WINNINGS>[{NUM}]+)\s?(?P<WCURRENCY2>[{LS}])?)? 

62 """.format(**substitutions), 

63 re.VERBOSE | re.MULTILINE | re.DOTALL, 

64 ) 

65 

66 re_category = re.compile( 

67 """ 

68 (?P<LIMIT>No\\sLimit|Fix\\sLimit|Pot\\sLimit)\\s 

69 (?P<GAME>Holdem|Omaha|OmahaHL|Hold\'em|Omaha\\sHi/Lo|OmahaHL) 

70 """.format(), 

71 re.VERBOSE | re.MULTILINE | re.DOTALL, 

72 )

74 codepage = ("utf8", "cp1252")

83 def parseSummary(self) -> None: # noqa: PLR0915, PLR0912, C901 

84 """Parse tournament summary text.""" 

85 m = self.re_tourney_info.search(self.summaryText) 

86 if m is None: 

87 tmp = self.summaryText[0:200] 

88 log.error("parseSummary failed: '%s'", tmp) 

89 raise FpdbParseError 

90 

91 mg = m.groupdict() 

92 

93 self.tourNo = mg["TOURNO"] 

94 

95 m1 = self.re_category.search(self.in_path) 

96 if m1: 

97 mg1 = m1.groupdict() 

98 if "LIMIT" in mg1 and mg1["LIMIT"] is not None: 

99 self.gametype["limitType"] = self.limits[mg1["LIMIT"]] 

100 else: 

101 self.gametype["limitType"] = "fl" 

102 if "GAME" in mg1: 

103 self.gametype["category"] = self.games[mg1["GAME"]][1] 

104 else: 

105 self.gametype["limitType"] = "nl" 

106 self.gametype["category"] = "holdem" 

107 

108 if "BUYIN" in mg and mg["BUYIN"] is not None: 

109 if mg["BUYIN"] == "Free" or mg["BIAMT"] is None: 

110 self.buyin = 0 

111 self.fee = 0 

112 else: 

113 self.buyin = int(100 * self.convert_to_decimal(mg["BIAMT"])) 

114 if mg["BIRAKE"] is None: 

115 self.fee = 0 

116 else: 

117 self.fee = int(100 * self.convert_to_decimal(mg["BIRAKE"])) 

118 

119 self.entries = mg["ENTRIES"] 

120 self.prizepool = self.buyin * int(self.entries) 

121 if "REBUYAMT" in mg and mg["REBUYAMT"] is not None: 

122 self.isRebuy = True 

123 self.rebuyCost = int(100 * self.convert_to_decimal(mg["REBUYAMT"])) 

124 if "ADDON" in mg and mg["ADDON"] is not None: 

125 self.isAddOn = True 

126 self.addOnCost = int(100 * self.convert_to_decimal(mg["ADDON"])) 

127 

128 if mg.get("CURRENCY1"): 

129 currency = mg["CURRENCY1"] 

130 elif mg.get("CURRENCY2"): 

131 currency = mg["CURRENCY2"] 

132 else: 

133 currency = None 

134 

135 if currency: 

136 if currency == "$": 

137 self.buyinCurrency = "USD" 

138 elif currency == "€": 

139 self.buyinCurrency = "EUR" 

140 elif self.buyin == 0: 

141 self.buyinCurrency = "FREE" 

142 else: 

143 self.buyinCurrency = "play" 

144 self.currency = self.buyinCurrency 

145 

146 player = mg["PNAME"] 

147 rank = int(mg["RANK"]) 

148 winnings = 0 

149 rebuy_count = None 

150 add_on_count = None 

151 ko_count = None 

152 

153 if "WINNINGS" in mg and mg["WINNINGS"] is not None: 

154 winnings = int(100 * self.convert_to_decimal(mg["WINNINGS"])) 

155 if mg.get("WCURRENCY"): 

156 if mg["WCURRENCY"] == "$": 

157 self.currency = "USD" 

158 elif mg["WCURRENCY"] == "€": 

159 self.currency = "EUR" 

160 elif mg.get("WCURRENCY2"): 

161 if mg["WCURRENCY2"] == "$": 

162 self.currency = "USD" 

163 elif mg["WCURRENCY2"] == "€": 

164 self.currency = "EUR" 

165 if "PREBUYS" in mg and mg["PREBUYS"] is not None: 

166 rebuy_count = int(mg["PREBUYS"]) 

167 if "PADDONS" in mg and mg["PADDONS"] is not None: 

168 add_on_count = int(mg["PADDONS"]) 

169 

170 self.addPlayer( 

171 rank, player, winnings, self.currency, rebuy_count, add_on_count, ko_count, 

172 )
