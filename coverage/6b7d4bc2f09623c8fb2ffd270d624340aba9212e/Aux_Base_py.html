36def clamp_to_screen(x: int, y: int, width: int = 200, height: int = 100) -> tuple[int, int]: 

37 """Clamp a window position to fit within the visible screen area. 

38 

39 Ensures that the given coordinates and window size do not extend beyond the boundaries of the screen. 

40 

41 Args: 

42 x: The x-coordinate of the window. 

43 y: The y-coordinate of the window. 

44 width: The width of the window (default is 200). 

45 height: The height of the window (default is 100). 

46 

47 Returns: 

48 tuple[int, int]: The clamped (x, y) coordinates within the screen bounds. 

49 """ 

50 from loggingFpdb import get_logger 

51 log = get_logger("hud") 

52 

53 app = QApplication.instance() 

54 if app is None: 

55 log.warning("No QApplication instance for screen clamping") 

56 return max(0, x), max(0, y) 

57 

58 # Try to find the screen containing the point 

59 screen = app.screenAt(QPoint(x, y)) 

60 if screen is None: 

61 # Fallback to primary screen 

62 screen = app.primaryScreen() 

63 log.info("Point (%d,%d) not on any screen, using primary screen", x, y) 

64 else: 

65 log.debug("Point (%d,%d) found on screen: %s", x, y, screen.name()) 

66 

67 if screen is None: 

68 log.warning("No screen available for clamping") 

69 return max(0, x), max(0, y) 

70 

71 geometry = screen.geometry() 

72 log.info("Screen geometry: X=%d, Y=%d, Width=%d, Height=%d", 

73 geometry.x(), geometry.y(), geometry.width(), geometry.height()) 

74 

75 # Clamp to the actual screen boundaries (including offset for extended screens) 

76 min_x = geometry.x() 

77 max_x = geometry.x() + geometry.width() - width 

78 min_y = geometry.y() 

79 max_y = geometry.y() + geometry.height() - height 

80 

81 clamped_x = max(min_x, min(x, max_x)) 

82 clamped_y = max(min_y, min(y, max_y)) 

83 

84 if clamped_x != x or clamped_y != y: 

85 log.info("CLAMPING: Original (%d,%d) -> Clamped (%d,%d) [Screen bounds: %d-%d, %d-%d]", 

86 x, y, clamped_x, clamped_y, min_x, max_x, min_y, max_y) 

87 

88 return clamped_x, clamped_y

382 def resize_windows(self) -> None: 

383 """Resize and reposition all HUD windows. 

384 

385 Updates the internal map of window positions based on the latest table and layout dimensions, 

386 then moves all windows accordingly. 

387 """ 

388 log.debug("RESIZING HUD WINDOWS - Table dimensions: %dx%d", self.hud.table.width, self.hud.table.height) 

389 # Resize calculation has already happened in HUD_main&hud.py 

390 # refresh our internal map to reflect these changes 

391 for i in list(range(1, self.hud.max + 1)): 

392 old_pos = self.positions.get(i, (0, 0)) 

393 self.positions[i] = self.hud.layout.location[self.adj[i]] 

394 log.debug("Seat %d position updated: (%d,%d) -> (%d,%d)", 

395 i, old_pos[0], old_pos[1], self.positions[i][0], self.positions[i][1]) 

396 old_common = self.positions.get("common", (0, 0)) 

397 self.positions["common"] = self.hud.layout.common 

398 log.debug("Common position updated: (%d,%d) -> (%d,%d)", 

399 old_common[0], old_common[1], self.positions["common"][0], self.positions["common"][1]) 

400 # and then move everything to the new places 

401 self.move_windows()

403 def move_windows(self) -> None: 

404 """Move all seat and common windows to their correct positions. 

405 

406 Calculates the absolute positions for each window based on the table's current coordinates and layout, 

407 clamps them to the screen, and moves the windows accordingly. 

408 """ 

409 # Ensure table coordinates are valid (not negative or off-screen) 

410 table_x = max(0, self.hud.table.x) if self.hud.table.x is not None else 50 

411 table_y = max(0, self.hud.table.y) if self.hud.table.y is not None else 50 

412 

413 log.debug("MOVING HUD WINDOWS - Table position: X=%d, Y=%d (from table.x=%s, table.y=%s)", 

414 table_x, table_y, self.hud.table.x, self.hud.table.y) 

415 

416 for i in list(range(1, self.hud.max + 1)): 

417 pos_x = self.positions[i][0] + table_x 

418 pos_y = self.positions[i][1] + table_y 

419 clamped_x, clamped_y = clamp_to_screen(pos_x, pos_y) 

420 log.debug("Moving seat %d window: Layout pos (%d,%d) + Table pos (%d,%d) = " 

421 "Final pos (%d,%d) -> Clamped (%d,%d)", 

422 i, self.positions[i][0], self.positions[i][1], table_x, table_y, 

423 pos_x, pos_y, clamped_x, clamped_y) 

424 self.m_windows[i].move(clamped_x, clamped_y) 

425 

426 common_x = self.hud.layout.common[0] + table_x 

427 common_y = self.hud.layout.common[1] + table_y 

428 clamped_common_x, clamped_common_y = clamp_to_screen(common_x, common_y) 

429 log.debug("Moving common window: Layout pos (%d,%d) + Table pos (%d,%d) = Final pos (%d,%d) -> Clamped (%d,%d)", 

430 self.hud.layout.common[0], self.hud.layout.common[1], table_x, table_y, common_x, common_y, 

431 clamped_common_x, clamped_common_y) 

432 self.m_windows["common"].move(clamped_common_x, clamped_common_y)

434 def create(self) -> None: 

435 """Create and initialize all seat and common windows for the HUD. 

436 

437 Sets up the window objects for each seat and the common area, 

438 positions them according to the current layout and table size, and displays them as needed. 

439 """ 

440 log.debug("=== AUX_BASE CREATE() METHOD CALLED ===") 

441 self.adj = self.adj_seats() 

442 self.m_windows = {} # windows to put the card/hud items in 

443 for i in [*list(range(1, self.hud.max + 1)), "common"]: 

444 if i == "common": 

445 # The common window is different from the others. Note that it needs to 

446 # get realized, shown, topified, etc. in create_common 

447 # self.hud.layout.xxxxx is updated here after scaling, to ensure 

448 # layout and positions are in sync 

449 (x, y) = self.hud.layout.common 

450 self.m_windows[i] = self.create_common(x, y) 

451 self.hud.layout.common = self.create_scale_position(x, y) 

452 else: 

453 (x, y) = self.hud.layout.location[self.adj[i]] 

454 log.debug("Seat %s: Loading position from layout: (%s, %s)", i, x, y) 

455 self.m_windows[i] = self.aw_class_window(self, i) 

456 self.positions[i] = self.create_scale_position(x, y) 

457 table_x = self.hud.table.x if self.hud.table.x is not None else 0 

458 table_y = self.hud.table.y if self.hud.table.y is not None else 0 

459 pos_x = max(0, self.positions[i][0] + table_x) 

460 pos_y = max(0, self.positions[i][1] + table_y) 

461 log.debug("=== AUX_BASE POSITIONING === Seat %s: table(%s, %s) + relative(%s, %s) = final(%s, %s)", 

462 i, table_x, table_y, self.positions[i][0], self.positions[i][1], pos_x, pos_y) 

463 self.m_windows[i].move(pos_x, pos_y) 

464 # Verify position after move 

465 actual_pos = self.m_windows[i].pos() 

466 log.debug("=== POSITION AFTER MOVE === Seat %s: requested(%s, %s) -> actual(%s, %s)", 

467 i, pos_x, pos_y, actual_pos.x(), actual_pos.y()) 

468 self.hud.layout.location[self.adj[i]] = self.positions[i] 

469 if "opacity" in self.params: 

470 self.m_windows[i].setWindowOpacity(float(self.params["opacity"])) 

471 

472 # main action below - fill the created window with content 

473 # the create_contents method is supplied by the subclass 

474 # for hud's this is probably Aux_Hud.stat_window 

475 self.create_contents(self.m_windows[i], i) 

476 

477 self.m_windows[i].create() # ensure there is a native window handle for topify 

478 log.debug("=== AUX_BASE CALLING TOPIFY === window[%d]=%s, table=%s", i, self.m_windows[i], 

479 self.hud.table.title if hasattr(self.hud.table, "title") else "NO_TITLE") 

480 self.hud.table.topify(self.m_windows[i]) 

481 if not self.uses_timer: 

482 self.m_windows[i].show() 

483 

484 self.hud.layout.height = self.hud.table.height 

485 self.hud.layout.width = self.hud.table.width

487 def create_scale_position(self, x: int, y: int) -> tuple[int, int]: 

488 """Scale a position according to the current table size. 

489 

490 For a given x/y, scale according to current height/width vs. reference 

491 height/width. This method is needed for create (because the table may not be 

492 the same size as the layout in config). 

493 

494 Any subsequent resizing of this table will be handled through 

495 hud_main.idle_resize. 

496 """ 

497 lw, lh = self.hud.layout.width, self.hud.layout.height 

498 

499 if lw == 0 or lh == 0: 

500 msg = "Layout width/height cannot be zero when scaling positions" 

501 raise ValueError(msg) 

502 

503 x_scale = self.hud.table.width / lw 

504 y_scale = self.hud.table.height / lh 

505 

506 scaled_x = int(x * x_scale) 

507 scaled_y = int(y * y_scale) 

508 

509 log.debug("=== SCALING DEBUG === Original(%d,%d) Layout(%dx%d) Table(%dx%d) Scale(%.2f,%.2f) Result(%d,%d)", 

510 x, y, lw, lh, self.hud.table.width, self.hud.table.height, x_scale, y_scale, scaled_x, scaled_y) 

511 

512 return scaled_x, scaled_y

560 def configure_event_cb(self, widget: SeatWindow, i: int | str) -> None: 

561 """Update the current location for each statblock. 

562 

563 This method is needed to record moves for an individual block. 

564 Move/resize also end up in here due to it being a configure. 

565 This is not optimal, but isn't easy to work around. fixme. 

566 """ 

567 if i: 

568 new_abs_position = widget.pos() # absolute value of the new position 

569 # Ensure table coordinates are valid for calculation 

570 table_x = self.hud.table.x if self.hud.table.x is not None else 0 

571 table_y = self.hud.table.y if self.hud.table.y is not None else 0 

572 new_position = ( 

573 new_abs_position.x() - table_x, 

574 new_abs_position.y() - table_y, 

575 ) 

576 log.debug("Seat %s: Position updated - abs(%s, %s) - table(%s, %s) = relative(%s, %s)", 

577 i, new_abs_position.x(), new_abs_position.y(), table_x, table_y, 

578 new_position[0], new_position[1]) 

579 self.positions[i] = new_position # write this back to our map 

580 if i != "common": 

581 self.hud.layout.location[self.adj[i]] = new_position # update the hud-level dict, 

582 # so other aux can be told 

583 else: 

584 self.hud.layout.common = new_position

586 def adj_seats(self) -> list[int]: 

587 """Determine how to adjust seating arrangements. 

588 

589 If a "preferred seat" is set in the hud layout configuration. 

590 Need range here, not xrange -> need the actual list. 

591 """ 

592 adj = list(range(self.hud.max + 1)) # default seat adjustments = no adjustment 

593 

594 # does the user have a fav_seat? if so, just get out now 

595 if self.hud.site_parameters["fav_seat"][self.hud.max] == 0: 

596 return adj 

597 

598 # find the hero's actual seat 

599 

600 actual_seat = None 

601 for key in self.hud.stat_dict: 

602 if self.hud.stat_dict[key]["screen_name"] == self.config.supported_sites[self.hud.site].screen_name: 

603 # Seat from stat_dict is the seat num recorded in the hand history and database 

604 # For tables <10-max, some sites omit some seat nums (e.g. iPoker 6-max uses 1,3,5,6,8,10) 

605 # The seat nums in the hh from the site are recorded in config file for each layout, and available 

606 # here as the self.layout.hh_seats list 

607 # (e.g. for iPoker - [None,1,3,5,6,8,10]; 

608 # for most sites- [None, 1,2,3,4,5,6] 

609 # we need to match 'seat' from hand history with the postion in the list, as the hud 

610 # always numbers its stat_windows using consecutive numbers (e.g. 1-6) 

611 

612 for i in range(1, self.hud.max + 1): 

613 if self.hud.layout.hh_seats[i] == self.hud.stat_dict[key]["seat"]: 

614 actual_seat = i 

615 break 

616 

617 if not actual_seat: # this shouldn't happen because we don't create huds if the hero isn't seated. 

618 log.error("Error finding hero seat.") 

619 return adj 

620 

621 for i in range(self.hud.max + 1): 

622 j = actual_seat + i 

623 if j > self.hud.max: 

624 j = j - self.hud.max 

625 adj[j] = self.hud.site_parameters["fav_seat"][self.hud.max] + i 

626 if adj[j] > self.hud.max: 

627 adj[j] = adj[j] - self.hud.max 

628 

629 return adj
