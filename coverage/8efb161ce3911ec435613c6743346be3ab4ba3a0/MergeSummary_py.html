350 def parseSummaryFile(self): 

351 self.buyinCurrency = "USD" 

352 soup = BeautifulSoup(self.summaryText) 

353 tables = soup.findAll("table") 

354 if len(tables) > 1: 

355 table1 = BeautifulSoup(str(tables[0])).findAll("tr") 

356 table2 = BeautifulSoup(str(tables[1])).findAll("tr") 

357 # FIXME: Searching every line for all regexes is pretty horrible 

358 # FIXME: Need to search for 'Status: Finished' 

359 # print self.in_path 

360 for p in table1: 

361 m = self.re_HTMLGameType.search(str(p)) 

362 if m: 

363 # print "DEBUG: re_HTMLGameType: '%s' '%s'" %(m.group('LIMIT'), m.group('GAME')) 

364 if m.group("GAME").strip() in self.mixes: 

365 self.gametype["category"] = self.mixes[m.group("GAME").strip()] 

366 else: 

367 self.gametype["category"] = self.games_html[m.group("GAME").strip()][1] 

368 self.gametype["limitType"] = self.limits[m.group("LIMIT").strip()] 

369 m = self.re_HTMLTourNo.search(str(p)) 

370 if m: 

371 # print "DEBUG: re_HTMLTourNo: '%s'" % m.group('TOURNO') 

372 self.tourNo = m.group("TOURNO").strip() 

373 m = self.re_HTMLName.search(str(p)) 

374 if m: 

375 # print "DEBUG: re_HTMLName: '%s'" % m.group('NAME') 

376 self.tourneyName = m.group("NAME").strip()[:40] 

377 if m.group("NAME").find("$") != -1: 

378 self.buyinCurrency = "USD" 

379 elif m.group("NAME").find("€") != -1: 

380 self.buyinCurrency = "EUR" 

381 m = self.re_HTMLPrizepool.search(str(p)) 

382 if m: 

383 # print "DEBUG: re_HTMLPrizepool: '%s'" % m.group('PRIZEPOOL') 

384 self.prizepool = int(self.convert_to_decimal(m.group("PRIZEPOOL").strip())) 

385 m = self.re_HTMLBuyIn.search(str(p)) 

386 if m: 

387 # print "DEBUG: re_HTMLBuyIn: '%s'" % m.group('BUYIN') 

388 self.buyin = int(100 * self.convert_to_decimal(m.group("BUYIN").strip())) 

389 if self.buyin == 0: 

390 self.buyinCurrency = "FREE" 

391 m = self.re_HTMLFee.search(str(p)) 

392 if m: 

393 # print "DEBUG: re_HTMLFee: '%s'" % m.group('FEE') 

394 self.fee = int(100 * self.convert_to_decimal(m.group("FEE").strip()))
