266 def walkDirectory(self, dir, sitelist) -> None: 

267 """Walks a directory, and executes a callback on each file.""" 

268 dir = os.path.abspath(dir) 

269 for file in [file for file in os.listdir(dir) if not file in [".", ".."]]: 

270 nfile = os.path.join(dir, file) 

271 if os.path.isdir(nfile): 

272 self.walkDirectory(nfile, sitelist) 

273 else: 

274 self.processFile(nfile)

297 def read_file(self, in_path): 

298 # Ignore macOS-specific hidden files such as .DS_Store 

299 if in_path.endswith(".DS_Store"): 

300 log.warning(f"Skipping system file {in_path}") 

301 return None, None 

302 

303 # Excel file management if xlrd is available 

304 if (in_path.endswith((".xls", ".xlsx"))) and xlrd: 

305 try: 

306 wb = xlrd.open_workbook(in_path) 

307 sh = wb.sheet_by_index(0) 

308 header = str(sh.cell(0, 0).value) 

309 return header, "utf-8" 

310 except (OSError, xlrd.XLRDError) as e: 

311 log.exception(f"Error reading Excel file {in_path}: {e}") 

312 return None, None 

313 

314 # Check for the presence of a BOM for UTF-16 

315 try: 

316 with open(in_path, "rb") as infile: 

317 raw_data = infile.read() 

318 

319 # If the file begins with a UTF-16 BOM (little endian or big endian) 

320 if raw_data.startswith((b"\xff\xfe", b"\xfe\xff")): 

321 try: 

322 whole_file = raw_data.decode("utf-16") 

323 return whole_file, "utf-16" 

324 except UnicodeDecodeError as e: 

325 log.exception(f"Error decoding UTF-16 file {in_path}: {e}") 

326 return None, None 

327 except OSError as e: 

328 log.exception(f"Error reading file {in_path}: {e}") 

329 return None, None 

330 

331 # Try different encodings in the `self.codepage` list 

332 for kodec in self.codepage: 

333 try: 

334 with codecs.open(in_path, "r", kodec) as infile: 

335 whole_file = infile.read() 

336 return whole_file, kodec 

337 except (OSError, UnicodeDecodeError) as e: 

338 log.warning(f"Failed to read file {in_path} with codec {kodec}: {e}") 

339 continue 

340 

341 log.error(f"Unable to read file {in_path} with any known codecs.") 

342 return None, None

344 def idSite(self, path, whole_file, kodec): 

345 """Identifies the site the hh file originated from.""" 

346 f = FPDBFile(path) 

347 f.kodec = kodec 

348 # DEBUG:print('idsite path',path ) 

349 # DEBUG:print('idsite f',f,f.ftype,f.site,f.gametype ) 

350 

351 # DEBUG:print('idsite self.sitelist.items',self.sitelist.items()) 

352 for _id, site in list(self.sitelist.items()): 

353 filter_name = site.filter_name 

354 m = site.re_Identify.search(whole_file[:5000]) 

355 if m and filter_name in ("Fulltilt", "PokerStars"): 

356 m1 = re_Divider[filter_name].search(whole_file.replace("\r\n", "\n")) 

357 if m1: 

358 f.archive = True 

359 f.archiveDivider = True 

360 elif re_Head.get(filter_name) and re_Head[filter_name].match( 

361 whole_file[:5000].replace("\r\n", "\n"), 

362 ): 

363 f.archive = True 

364 f.archiveHead = True 

365 if m: 

366 # For PokerStars, we need to determine the specific skin 

367 if filter_name == "PokerStars": 

368 # Temporary import of the module to detect the skin 

369 from PokerStarsToFpdb import PokerStars 

370 

371 ps = PokerStars(self.config, in_path=path, autostart=False) 

372 skin_name, skin_id = ps.detectPokerStarsSkin(whole_file, path) 

373 

374 # Search for the site corresponding to the skin detected 

375 for _sid, s in list(self.sitelist.items()): 

376 if s.name == skin_name: 

377 f.site = s 

378 break 

379 else: 

380 # If we can't find the specific skin, we'll use the default site 

381 f.site = site 

382 else: 

383 f.site = site 

384 

385 f.ftype = "hh" 

386 if f.site.re_HeroCards: 

387 h = f.site.re_HeroCards.search(whole_file[:5000]) 

388 if h and "PNAME" in h.groupdict(): 

389 f.hero = h.group("PNAME") 

390 else: 

391 f.hero = "Hero" 

392 return f 

393 

394 for _id, site in list(self.sitelist.items()): 

395 if site.summary: 

396 if path.endswith((".xls", ".xlsx")): 

397 filter_name = site.filter_name 

398 if filter_name in ("Fulltilt", "PokerStars"): 

399 m2 = re_XLS[filter_name].search(whole_file[:5000]) 

400 if m2: 

401 f.site = site 

402 f.ftype = "summary" 

403 return f 

404 else: 

405 m3 = site.re_SumIdentify.search(whole_file[:10000]) 

406 if m3: 

407 f.site = site 

408 f.ftype = "summary" 

409 return f 

410 

411 m1 = self.re_Identify_PT.search(whole_file[:5000]) 

412 m2 = self.re_SumIdentify_PT.search(whole_file[:100]) 

413 if m1 or m2: 

414 filter = "PokerTrackerToFpdb" 

415 filter_name = "PokerTracker" 

416 mod = __import__(filter) 

417 obj = getattr(mod, filter_name, None) 

418 summary = "PokerTrackerSummary" 

419 

420 # Detect specific iPoker skin for PokerTracker format 

421 detected_site_name = "PokerTracker" # default 

422 if m1 and "GAME #" in m1.group(): 

423 # This is an iPoker hand, detect the specific skin 

424 detected_site_name = self.detectiPokerSkin(whole_file) 

425 log.debug(f"Detected iPoker skin from PokerTracker format: {detected_site_name}") 

426 

427 f.site = Site(detected_site_name, filter, filter_name, summary, obj) 

428 if m1: 

429 f.ftype = "hh" 

430 if re.search(r"\*{2}\sGame\sID\s", m1.group()): 

431 f.site.line_delimiter = None 

432 f.site.re_SplitHands = re.compile(r"End\sof\sgame\s\d+") 

433 elif re.search(r"\*{2}\sHand\s\#\s", m1.group()): 

434 f.site.line_delimiter = None 

435 f.site.re_SplitHands = re.compile(r"Rake:\s[^\s]+") 

436 elif re.search(r"Server\spoker\d+\.ipoker\.com", whole_file[:250]): 

437 f.site.line_delimiter = None 

438 f.site.spaces = True 

439 f.site.re_SplitHands = re.compile(r"GAME\s\#") 

440 m3 = f.site.re_HeroCards1.search(whole_file[:5000]) 

441 if m3: 

442 f.hero = m3.group("PNAME") 

443 else: 

444 m4 = f.site.re_HeroCards2.search(whole_file[:5000]) 

445 if m4: 

446 f.hero = m4.group("PNAME") 

447 else: 

448 f.ftype = "summary" 

449 return f 

450 

451 return False
