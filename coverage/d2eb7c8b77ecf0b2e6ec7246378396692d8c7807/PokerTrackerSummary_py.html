32 """PokerTracker tournament summary parser."""

34 hhtype: ClassVar = "summary" 

35 limits: ClassVar = { 

36 "NL": "nl", 

37 "No Limit": "nl", 

38 "Pot Limit": "pl", 

39 "PL": "pl", 

40 "FL": "fl", 

41 "Limit": "fl", 

42 "LIMIT": "fl", 

43 } 

44 games: ClassVar = { # base, category 

45 "Hold'em": ("hold", "holdem"), 

46 "Texas Hold'em": ("hold", "holdem"), 

47 "Holdem": ("hold", "holdem"), 

48 "Omaha": ("hold", "omahahi"), 

49 "Omaha Hi": ("hold", "omahahi"), 

50 "Omaha Hi/Lo": ("hold", "omahahilo"),

53 substitutions: ClassVar = { 

54 "LEGAL_ISO": "USD|EUR|GBP|CAD|FPP", # legal ISO currency codes 

55 "LS": r"\$|€|\£|P|SC|", # legal currency symbols - Euro(cp1252, utf-8) 

56 "PLYR": r"(?P<PNAME>.+?)", 

57 "NUM": r".,\d", 

58 "CUR": r"(\$|€||\£|)",

61 re_identify = re.compile("PokerTracker")

63 re_tourney_info = re.compile( 

64 """ 

65 \\s(3|4)\\sTournament\\sSummary\\s+ 

66 Site:\\s(?P<SITE>.+?)\\s+ 

67 Game:\\s(?P<GAME>Holdem|Texas\\sHold\'em|Omaha|Omaha\\sHi|Omaha\\sHi\\/Lo)\\s+ 

68 Tournament\\s\\#:\\s(?P<TOURNO>[0-9]+)\\s+ 

69 Started:\\s(?P<DATETIME>.+?)\\s+ 

70 Finished:\\s(?P<DATETIME1>.+?)\\s+ 

71 Buyin:\\s(?P<CURRENCY>[{LS}]?)(?P<BUYIN>[,.0-9]+)\\s+ 

72 (Bounty:\\s[{LS}]?(?P<BOUNTY>[,.0-9]+)\\s+)? 

73 Fee:\\s[{LS}]?(?P<FEE>[,.0-9]+)\\s+ 

74 (Prize\\sPool:\\s[{LS}]?(?P<PRIZEPOOL>[,.0-9]+)\\s+)? 

75 (Rebuy:\\s[{LS}]?(?P<REBUYAMT>[,.0-9]+)\\s+)? 

76 (Addon:\\s[{LS}]?(?P<ADDON>[,.0-9]+)\\s+)? 

77 Initial\\sStack:\\s(?P<STACK>[0-9]+)\\s+ 

78 Table\\sType:\\s(?P<TYPE>.+?)\\s+ 

79 Tourney\\sType:\\s(?P<LIMIT>No\\sLimit|Limit|LIMIT|Pot\\sLimit|N\\/A)\\s+ 

80 Players:\\s(?P<ENTRIES>\\d+)\\s+ 

81 """.format(**substitutions), 

82 re.VERBOSE | re.MULTILINE, 

83 ) 

84 

85 re_max = re.compile(r"\((?P<MAX>\d+)\smax\)\s") 

86 re_speed = re.compile(r"(?P<SPEED>Turbo|Hyper\-Turbo)") 

87 re_sng = re.compile(r"\s(?P<SNG>SNG)\s") 

88 

89 re_player = re.compile( 

90 r""" 

91 Place:\s(?P<RANK>[0-9]+),\s 

92 Player:\s(?P<NAME>.*),\s 

93 Won:\s(?P<CUR>[{LS}]?)(?P<WINNINGS>[,.0-9]+), 

94 (\sBounties:\s(?P<KOS>\d+),)? 

95 (\sRebuys:\s(?P<REBUYS>\d+),)? 

96 (\sAddons:\s(?P<ADDONS>\d+),)? 

97 """.format(**substitutions), 

98 re.VERBOSE, 

99 ) 

100 

101 re_date_time = re.compile( 

102 r"""(?P<Y>[0-9]{4})\/(?P<M>[0-9]{2})\/(?P<D>[0-9]{2})[\- ]+(?P<H>[0-9]+):(?P<MIN>[0-9]+):(?P<S>[0-9]+)""", 

103 re.MULTILINE, 

104 )

123 def parseSummary(self) -> None: # noqa: PLR0912, PLR0915, C901 

124 """Parse tournament summary text.""" 

125 m = self.re_tourney_info.search(self.summaryText) 

126 if m is None: 

127 tmp = self.summaryText[0:200] 

128 log.error("parse Summary failed: '%s'", tmp) 

129 raise FpdbParseError 

130 

131 mg = m.groupdict() 

132 if "SITE" in mg: 

133 if self.site_name_map.get(mg["SITE"]) is not None: 

134 self.siteName = self.site_name_map.get(mg["SITE"]) 

135 self.siteId = self.SITEIDS.get(self.siteName) 

136 else: 

137 tmp = self.summaryText[0:200] 

138 log.error("Unsupported site summary '%s'", tmp) 

139 raise FpdbParseError 

140 if "TOURNO" in mg: 

141 self.tourNo = mg["TOURNO"] 

142 if "GAME" in mg: 

143 self.gametype["category"] = self.games[mg["GAME"]][1] 

144 if mg["LIMIT"] in self.limits: 

145 self.gametype["limitType"] = self.limits[mg["LIMIT"]] 

146 elif self.gametype["category"] == "holdem": 

147 self.gametype["limitType"] = "nl" 

148 else: 

149 self.gametype["limitType"] = "pl" 

150 if "TYPE" in mg: 

151 self.tourneyName = mg["TYPE"] 

152 t1 = self.re_max.search(self.tourneyName) 

153 if t1: 

154 self.maxseats = int(t1.group("MAX")) 

155 t2 = self.re_speed.search(self.tourneyName) 

156 if t2: 

157 if t2.group("SPEED") == "Turbo": 

158 self.speed = "Turbo" 

159 elif t2.group("SPEED") == "Hyper-Turbo": 

160 self.speed = "Hyper" 

161 t3 = self.re_sng.search(self.tourneyName) 

162 if t3: 

163 self.isSng = True 

164 if "DEEP" in self.tourneyName: 

165 self.stack = "Deep" 

166 if "SATELLITE" in self.tourneyName: 

167 self.isSatellite = True 

168 if mg["BUYIN"] is not None: 

169 self.buyin = int(100 * float(self.clearMoneyString(mg["BUYIN"]))) 

170 if mg["FEE"] is not None: 

171 self.fee = int(100 * float(self.clearMoneyString(mg["FEE"]))) 

172 if "REBUYAMT" in mg and mg["REBUYAMT"] is not None: 

173 self.isRebuy = True 

174 self.rebuyCost = int(100 * float(self.clearMoneyString(mg["REBUYAMT"]))) 

175 if "PRIZEPOOL" in mg and mg["PRIZEPOOL"] is not None: 

176 self.prizepool = int(100 * float(self.clearMoneyString(mg["PRIZEPOOL"]))) 

177 if "ADDON" in mg and mg["ADDON"] is not None: 

178 self.isAddOn = True 

179 self.addOnCost = int(100 * float(self.clearMoneyString(mg["ADDON"]))) 

180 if "BOUNTY" in mg and mg["BOUNTY"] is not None: 

181 self.koBounty = int(100 * float(self.clearMoneyString(mg["BOUNTY"]))) 

182 self.isKO = True 

183 if "ENTRIES" in mg: 

184 self.entries = mg["ENTRIES"] 

185 if "DATETIME" in mg: 

186 m1 = self.re_date_time.finditer(mg["DATETIME"]) 

187 for a in m1: 

188 datetimestr = "{}/{}/{} {}:{}:{}".format( 

189 a.group("Y"), 

190 a.group("M"), 

191 a.group("D"), 

192 a.group("H"), 

193 a.group("MIN"), 

194 a.group("S"), 

195 ) 

196 else: 

197 datetimestr = "2000/01/01 00:00:00" # default used if time not found 

198 

199 self.startTime = datetime.datetime.strptime( 

200 datetimestr, "%Y/%m/%d %H:%M:%S", 

201 ).replace(tzinfo=datetime.timezone.utc) # also timezone at end, e.g. " ET" 

202 

203 if mg["CURRENCY"] == "$": 

204 self.buyinCurrency = "USD" 

205 elif mg["CURRENCY"] == "€": 

206 self.buyinCurrency = "EUR" 

207 elif mg["CURRENCY"] in ("SC", "P"): 

208 self.buyinCurrency = "PSFP" 

209 elif not mg["CURRENCY"]: 

210 self.buyinCurrency = "play" 

211 if self.buyin == 0: 

212 self.buyinCurrency = "FREE" 

213 self.currency = self.buyinCurrency 

214 

215 if ( 

216 self.buyinCurrency not in ("FREE", "PSFP") 

217 and "ENTRIES" in mg 

218 and self.prizepool == 0 

219 ): 

220 self.prizepool = int(Decimal(self.clearMoneyString(mg["BUYIN"]))) * int( 

221 self.entries, 

222 ) 

223 

224 m = self.re_player.finditer(self.summaryText) 

225 for a in m: 

226 mg = a.groupdict() 

227 name = mg["NAME"] 

228 rank = int(mg["RANK"]) 

229 winnings = 0 

230 rebuy_count = None 

231 add_on_count = None 

232 ko_count = None 

233 if len(name) > 0: 

234 if "WINNINGS" in mg and mg["WINNINGS"] is not None: 

235 winnings = int(100 * float(self.clearMoneyString(mg["WINNINGS"]))) 

236 

237 if "REBUYS" in mg and mg["REBUYS"] is not None: 

238 rebuy_count = int(mg["REBUYS"]) 

239 

240 if "ADDONS" in mg and mg["ADDONS"] is not None: 

241 add_on_count = int(mg["ADDONS"]) 

242 

243 if "KOS" in mg and mg["KOS"] is not None: 

244 ko_count = int(mg["KOS"]) 

245 

246 if "CUR" in mg and mg["CUR"] is not None: 

247 if mg["CUR"] == "$": 

248 self.currency = "USD" 

249 elif mg["CUR"] == "€": 

250 self.currency = "EUR" 

251 elif mg["CUR"] in ("P", "SC"): 

252 self.currency = "PSFP" 

253 

254 if rank == 0: 

255 rank = None 

256 winnings = None 

257 

258 if len(name) == 0: 

259 log.debug("a.groupdict(): %s", mg) 

260 

261 self.addPlayer( 

262 rank, name, winnings, self.currency, rebuy_count, add_on_count, ko_count, 

263 )
